CREATE TABLE PELICULAS(
ID NUMBER PRIMARY KEY,
NOMBRE VARCHAR2(70) NOT NULL UNIQUE,
DURACION NUMBER NOT NULL,--en minutos
FECHA_LANZ DATE NOT NULL
);

CREATE SEQUENCE SECU_PELIS_ID
START WITH 1 
INCREMENT  BY 1;


CREATE OR REPLACE TRIGGER SEC
BEFORE INSERT
ON PELICULAS
FOR EACH ROW
BEGIN
SELECT SECU_PELIS_ID.NEXTVAL INTO :NEW.ID FROM DUAL;
END;
